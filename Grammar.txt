// opening the file
//
Opening : Begin PRP | ExtendBegin PRP
Begin : Beginfile Class
Beginfile : beginfile
Class : class

Report : report
N : n

// Imports

Imports : Bimport Punc
Bimport : Import Iname
Import : import
Iname : iName
//
// Closing
//
Closing : Punc FinalEnd
FinalEnd : Signee Punc
Signee : signee

// print statements - to be updated in milestone 3

Print : print
Tempstr : strLit | string | true | false | boolType | numType | charLit | charType | dubLit

// Variable declaration

Declaration : decType Punc
decType : VarDec FullType

varType : strType | boolType | numType | charType
asignDec : Asign matchedLit
Type : Bool | double | char | string
matchedLit : Bool true | Bool false | Bool null | Double Null | Double DubLit | Char Null | Char CharLit | String StrLit | String Null
Null : null
Double : double
Char : char
String : string
bool : Bool
True : true
False : false
DubLit : dubLit
